此函数的原型为char *strrev(char *str)
功能为反转字符串，返回字符串指针。
    char *strrev(char *str)  
    {  
        if(str==NULL)  
            return NULL;  
        char *start=str;  
        char *end=str+strlen(str)-1;  
        char temp;  
        while(start<end)  
        {  
            temp=*start;  
            *start=*end;  
            *end=temp;  
            start++;  
            end--;  
        }  
        return str;  
    }
